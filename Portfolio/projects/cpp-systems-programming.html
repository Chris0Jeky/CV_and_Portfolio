Thread-Safe Search with Multiple Indexes

std::vector<TrackPtr> MusicLibrary::search(const std::string& query) const {
    std::shared_lock lock(mutex_);  // Multiple readers allowed
    
    std::vector<TrackPtr> results;
    std::unordered_set<TrackPtr> unique_results;
    
    // Search across all indexes
    auto search_in_index = [&](const auto& index) {
        auto range = index.equal_range(query);
        for (auto it = range.first; it != range.second; ++it) {
            if (it->second < tracks_.size() && tracks_[it->second]) {
                unique_results.insert(tracks_[it->second]);
            }
        }
    };
    
    search_in_index(title_index_);
    search_in_index(artist_index_);
    search_in_index(album_index_);
    search_in_index(genre_index_);
    
    results.assign(unique_results.begin(), unique_results.end());
    return results;
}

Fuzzy Search with Levenshtein Distance

size_t SearchEngine::levenshtein_distance(const std::string& s1, const std::string& s2) {
    const size_t m = s1.size();
    const size_t n = s2.size();
    
    if (m == 0) return n;
    if (n == 0) return m;
    
    std::vector<std::vector<size_t>> dp(m + 1, std::vector<size_t>(n + 1));
    
    // Initialize base cases
    for (size_t i = 0; i <= m; ++i) dp[i][0] = i;
    for (size_t j = 0; j <= n; ++j) dp[0][j] = j;
    
    // Fill the matrix
    for (size_t i = 1; i <= m; ++i) {
        for (size_t j = 1; j <= n; ++j) {
            size_t cost = (s1[i-1] == s2[j-1]) ? 0 : 1;
            
            dp[i][j] = std::min({
                dp[i-1][j] + 1,      // deletion
                dp[i][j-1] + 1,      // insertion
                dp[i-1][j-1] + cost  // substitution
            });
        }
    }
    
    return dp[m][n];
}

CSV Parser with Quoted Field Support

std::vector<std::string> FileIO::parse_csv_line(const std::string& line) {
    std::vector<std::string> fields;
    std::string current_field;
    bool in_quotes = false;
    
    for (size_t i = 0; i < line.length(); ++i) {
        char c = line[i];
        
        if (in_quotes) {
            if (c == '"' && i + 1 < line.length() && line[i + 1] == '"') {
                current_field += '"';  // Escaped quote
                ++i;
            } else if (c == '"') {
                in_quotes = false;     // End of quoted field
            } else {
                current_field += c;
            }
        } else {
            if (c == '"' && current_field.empty()) {
                in_quotes = true;      // Start of quoted field
            } else if (c == ',') {
                fields.push_back(current_field);
                current_field.clear();
            } else {
                current_field += c;
            }
        }
    }
    
    fields.push_back(current_field);  // Don't forget the last field
    return fields;
}
